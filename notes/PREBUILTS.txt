You need to build libarchive
git clone -b v3.8.4 https://github.com/libarchive/libarchive.git
cd libarchive
mkdir build_arafa
cd build_arafa
cmake .. \
  -DCMAKE_BUILD_TYPE=Release \
  -DENABLE_TEST=OFF \
  -DENABLE_CAT=OFF \
  -DENABLE_TAR=OFF \
  -DENABLE_CPIO=OFF \
  -DENABLE_INSTALL=ON

make -j16
mkdir install
make install DESTDIR=./install
